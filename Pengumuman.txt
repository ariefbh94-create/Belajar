<script>

  // GANTI DENGAN URL WEB APP APPS SCRIPT
  const API_URL = 'https://script.google.com/macros/s/AKfycbxnmwtVrCXyYHkTix4kUje4wJs9lcQ8-7UNrQkfWpAg00w6aGX6-5owtrVodAbp_9VG/exec';

  async function cekKelulusan() {
    const nis = document.getElementById('nis').value.trim();
    const result = document.getElementById('result');

    if (!nis) {
      alert('Masukkan NIS terlebih dahulu');
      return;
    }

    result.style.display = 'block';
    result.innerHTML = 'Loading...';

    try {
      const response = await fetch(`${API_URL}?nis=${nis}`);
      const data = await response.json();

      if (data.length > 0) {
        const siswa = data[0];

        result.innerHTML = `

          <h2>Data Kelulusan</h2>

          <div class="item">
            <span class="label">Kelas</span>
            ${siswa.kelas}
          </div>

          <div class="item">
            <span class="label">NIS</span>
            ${siswa.nis}
          </div>

          <div class="item">
            <span class="label">NISN</span>
            ${siswa.nisn}
          </div>

          <div class="item">
            <span class="label">Nama</span>
            ${siswa.nama}
          </div>

          <div class="status-lulus">
            ${siswa.keterangan}
          </div>
        `;

      } else {
        result.innerHTML = `
          <div class="not-found">
            Data tidak ditemukan
          </div>
        `;
      }

    } catch (error) {
      result.innerHTML = `
        <div class="not-found">
          Terjadi kesalahan koneksi
        </div>
      `;
    }
  }
  </script>
